1. Initialize with: `npx create-next-app@latest . --typescript --eslint --app --tailwind --src-dir --import-alias "@/*"`
   - If prompted about existing files, choose to continue (project dir exists but is mostly empty)

2. Install Tailwind v4 dependencies:
   ```bash
   npm install tailwindcss @tailwindcss/postcss postcss
   ```

3. Create `postcss.config.mjs`:
   ```javascript
   const config = {
     plugins: {
       '@tailwindcss/postcss': {},
     },
   }
   export default config
   ```

4. Update `next.config.js` for static export compatibility (needed for Hostinger deployment):
   ```javascript
   /** @type {import('next').NextConfig} */
   const nextConfig = {
     output: 'export',
     trailingSlash: true,
     images: {
       unoptimized: true,
     },
   }

   module.exports = nextConfig
   ```

5. Clean up the default `src/app/page.tsx` to a minimal placeholder:
   ```tsx
   export default function Home() {
     return (
       <main className="min-h-screen flex items-center justify-center">
         <h1 className="font-display text-4xl">MadMedia</h1>
       </main>
     )
   }
   ```

Note: Do NOT keep any default Next.js styling, gradients, or template content. This is a clean slate.
  </action>

<task type="auto">
  <name>Task 2: Configure fonts and design tokens</name>
  <files>
    - src/lib/fonts.ts
    - src/styles/globals.css
    - src/app/layout.tsx
  </files>
  <action>
1. Create `src/lib/fonts.ts` with character fonts (NOT generic):
   ```typescript
   import { Fraunces, DM_Sans } from 'next/font/google'

   export const fraunces = Fraunces({
     subsets: ['latin'],
     display: 'swap',
     variable: '--font-fraunces',
     axes: ['SOFT', 'WONK', 'opsz'],
   })

   export const dmSans = DM_Sans({
     subsets: ['latin'],
     display: 'swap',
     variable: '--font-dm-sans',
     weight: ['400', '500', '600', '700'],
   })
   ```

2. Replace `src/styles/globals.css` with Tailwind v4 @theme:
   ```css
   @import 'tailwindcss';

   @theme {
     /* Typography - Character fonts that don't look AI-generated */
     --font-display: var(--font-fraunces), Georgia, serif;
     --font-body: var(--font-dm-sans), system-ui, sans-serif;

     /* Colors - MadMedia brand palette */
     --color-primary: #1a1a1a;
     --color-primary-soft: #2d2d2d;
     --color-accent: #c9a227;
     --color-accent-light: #e8d48a;
     --color-accent-dark: #9a7b1c;
     --color-background: #fafafa;
     --color-background-warm: #f5f3ef;
     --color-text: #1a1a1a;
     --color-text-muted: #6b6b6b;

     /* Spacing - Dramatic whitespace (120px+) */
     --spacing-section: 7.5rem;
     --spacing-section-sm: 5rem;
     --spacing-section-lg: 10rem;

     /* Layout */
     --width-content: 75rem;

     /* Animation - Custom easing (NOT default ease or bounce) */
     --ease-out-expo: cubic-bezier(0.19, 1, 0.22, 1);
     --ease-out-quart: cubic-bezier(0.165, 0.84, 0.44, 1);
   }

   /* Base body styling */
   body {
     @apply font-body text-text bg-background antialiased;
   }
   ```

3. Update `src/app/layout.tsx` to wire fonts:
   ```tsx
   import type { Metadata } from 'next'
   import { fraunces, dmSans } from '@/lib/fonts'
   import '@/styles/globals.css'

   export const metadata: Metadata = {
     title: 'MadMedia - Social Media Management & Website Services',
     description: 'Professional social media management and website services for small businesses in Madison, MN and beyond.',
   }

   export default function RootLayout({
     children,
   }: {
     children: React.ReactNode
   }) {
     return (
       <html lang="en" className={`${fraunces.variable} ${dmSans.variable}`}>
         <body>{children}</body>
       </html>
     )
   }
   ```

4. Update the homepage `src/app/page.tsx` to demonstrate both fonts:
   ```tsx
   export default function Home() {
     return (
       <main className="min-h-screen flex flex-col items-center justify-center gap-4 px-6">
         <h1 className="font-display text-5xl md:text-7xl text-primary">
           MadMedia
         </h1>
         <p className="font-body text-lg text-text-muted max-w-md text-center">
           Social media management and website services for small businesses.
         </p>
         <div className="mt-8 h-1 w-24 bg-accent" />
       </main>
     )
   }
   ```
